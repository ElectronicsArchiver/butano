#define BTN_AUDIO_MIXING_RATE_8_KHZ
#include <btn_audio_mixing_rate.h>

8KHz mixing rate, provides poor quality.

#define BTN_AUDIO_MIXING_RATE_10_KHZ
#include <btn_audio_mixing_rate.h>

10Khz mixing rate, provides low quality.

#define BTN_AUDIO_MIXING_RATE_13_KHZ
#include <btn_audio_mixing_rate.h>

13Khz mixing rate, provides almost OK quality.

#define BTN_AUDIO_MIXING_RATE_16_KHZ
#include <btn_audio_mixing_rate.h>

16Khz mixing rate, provides OK quality, standard setting.

#define BTN_AUDIO_MIXING_RATE_18_KHZ
#include <btn_audio_mixing_rate.h>

18Khz mixing rate, provides better quality, high CPU load.

#define BTN_AUDIO_MIXING_RATE_21_KHZ
#include <btn_audio_mixing_rate.h>

21Khz mixing rate, provides even better quality, higher CPU load.

#define BTN_AUDIO_MIXING_RATE_27_KHZ
#include <btn_audio_mixing_rate.h>

27Khz mixing rate, provides almost the best quality, almost the highest CPU load.

#define BTN_AUDIO_MIXING_RATE_31_KHZ
#include <btn_audio_mixing_rate.h>

31Khz mixing rate, provides the best quality, highest CPU load.

#define BTN_CFG_AUDIO_MIXING_RATE
#include <btn_config_audio.h>

Specifies software audio mixing rate in KHz.

Values not specified in BTN_AUDIO_MIXING_RATE_* macros are not allowed.

#define BTN_CFG_AUDIO_MAX_COMMANDS
#include <btn_config_audio.h>

Specifies the size of the audio commands queue.

This queue is processed and cleared when btn::core::update() is called.
